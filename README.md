 #### mock store, action, getter
 - import { mount, createLocalVue } from '@vue/test-utils';
 - const localVue = createLocalVue();
 ```js
 import { mount, createLocalVue } from '@vue/test-utils';
 import Vuetify from 'vuetify';
 import Vuex from 'vuex';
 import Vue from 'vue';
 ...
 
 const localVue = createLocalVue();
 localVue.use(Vuex);
 
 describe('...', () => {
  let vuetify;
  let store;
  let actions;
  let getters;
  
  beforeEach(() => {
    vuetify = new Vuetify({});

    actions = {
      'test/index': jest.fn((item) => {
        return Promise.resolve({});
      }),
    };
    
    getters = {
      'test/getDataList': () => ({
        data: []
      }),
    };

    store = new Vuex.Store({
      state: {},
      actions,
      getters,
    });
  });
  
  test('...', async () => {
   const wrapper = mount(..., {
      localVue,
      store,
      vuetify,
      data: function() {
        return {}
      },
      template: `
        <div data-app>
          <... />
        </div>
      `,
      sync: false
  });
  await flushPromises();
  expect(actions['test/index']).toHaveBeenCalled();
 ```
